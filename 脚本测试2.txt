    end
    
    local startPos, endPos = string.find(chineseOption, "%((.+)%)$")
    if startPos and endPos then

    end
    
    for english, chinese in pairs(ChineseTranslations) do
        if chinese == chineseOption then
            return english
        end
    end
    
    return chineseOption
end
